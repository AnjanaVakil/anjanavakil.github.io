#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;
int main()
{
    // greedy
    int n, k;
    cin >> n >> k;
    vector<int> v(n);
    for (int i = 0; i < n; i++)
    {
        cin >> v[i];
    }
    sort(v.begin(), v.end());
    int res = 1;
    int last = v[0];
    for (int i = 1; i < n; i++)
    {
        if (v[i] - last <= k)
        {
            continue;
        }
        res++;
        last = v[i];
    }
    cout << res << endl;
    return 0;
}

static int read() {
  int x = 0, w = 1;
  char c = 0;
  while (c < '0' || c > '9') {
    if (c == '-') w = -1;
    c = getchar();
  }
  while (c >= '0' && c <= '9') {
    x = x * 10 + (c - '0');
    c = getchar();
  }
  return x * w;
}

{
    int n=read();
    int x=read();
    x=abs(x);
    vector<long long> a(N,0);

    for(int i=0;i<n;i++){
        int num=read();
        a[num+M]+=1;
    }
    if(x==0){
        for(int i=0;i<N;i++){
            if(a[i]==0) continue;
            res+=a[i]*a[i];
        }
    }else{
        for(int i=0;i<N-x;i++){
            res+=a[i]*a[i+x];
        }
    }
    printf("%lld",res);
    return 0;
}

    set<pair<int, int> > cache;
    int res = 0;
    int cnt = 0;
    for (int i = 1; i <= n; i++)
    {
        if (flag[v[i]])
        {
            // found, update
            auto it = cache.lower_bound(make_pair(i, v[i]));
            auto temp = *it;
            cache.erase(it);
            temp.first = nxt[i];
            cache.insert(temp);
            continue;
        }
        flag[v[i]] = 1;
        cnt++;
        res++;
        cache.insert(make_pair(nxt[i], v[i]));
        if (cnt > k)
        {
            cnt--;
            flag[(--cache.end())->second] = 0;
            cache.erase(--cache.end());
        }
    }
    printf("%d", res);
    return 0;
}

    v[t].next=head[a];
    head[a]=t;
}
void SPFA(){
    vis[0]=1;
    q.push(0);
    while(!q.empty()){
        int cur=q.front();
        q.pop();
        vis[cur]=0;
        if(tot[cur]==n-1){
            cout << -1;
            exit(0);

                dis[temp]=dis[cur]+v[i].w;
                if(!vis[temp]){
                    vis[temp]=1;
                    q.push(temp);
                }
            }
        }
    }
}
