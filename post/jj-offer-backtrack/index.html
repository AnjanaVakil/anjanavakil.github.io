class Solution {
public:
    vector<vector<int>> res;
    void backtrack(vector<int>& nums,int index){
        if(index==nums.size()){
            vector<int> temp;
            for(int num:nums){
                temp.push_back(num);
            }
            res.push_back(temp);
        }else{
            for(int j=index;j<nums.size();j++){
                swap(nums[index],nums[j]);
                backtrack(nums,index+1);
                swap(nums[index],nums[j]);
            }
        }
        
    }
    vector<vector<int>> permute(vector<int>& nums) {
        backtrack(nums,0);
        return res;
    }
};

class Solution {
public:
    vector<vector<int>> res;
    void backtrack(vector<int>& nums,int index){
        if(index==nums.size()){
            vector<int> temp;
            for(int num:nums){
                temp.push_back(num);
            }
            res.push_back(temp);
        }else{
            set<int> s;
            for(int j=index;j<nums.size();j++){
                if(s.find(nums[j])==s.end()){
                    s.insert(nums[j]);

                    swap(nums[index],nums[j]);
                    backtrack(nums,index+1);
                    swap(nums[index],nums[j]);
                }
            }
        }
    }
    vector<vector<int>> permuteUnique(vector<int>& nums) {
        backtrack(nums,0);
        return res;
    }
};
