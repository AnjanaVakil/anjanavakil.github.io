class EchoHandler implements Runnable {
    final SocketChannel channel;
    final SelectionKey sk;
    final ByteBuffer byteBuffer = ByteBuffer.allocate(1024);
    static final int RECIEVING = 0, SENDING = 1;
    int state = RECIEVING;

    EchoHandler(Selector selector, SocketChannel c) throws IOException {

                //å†™å…¥é€šé“
                channel.write(byteBuffer);

                state = RECIEVING;
            } else if (state == RECIEVING) {
                //ä»é€šé“è¯»

                    Logger.info(new String(byteBuffer.array(), 0, length));
                }
                //è¯»å®Œåï¼Œå‡†å¤‡å¼€å§‹å†™å…¥é€šé“,byteBufferåˆ‡æ¢æˆè¯»æ¨¡å¼
                byteBuffer.flip();

                sk.interestOps(SelectionKey.OP_WRITE);
                //è¯»å®Œå,è¿›å…¥å‘é€çš„çŠ¶æ€

        } catch (IOException ex) {

                channel.finishConnect();

                e.printStackTrace();
            }
        }
    }
    
}

        Reactor subReactor2 = new Reactor(workSelectors[1]);

    private void startService() {
        new Thread(bossReactor).start();
        // ä¸€å­ååº”å™¨å¯¹åº”ä¸€æ¡çº¿ç¨‹

    class Reactor implements Runnable {
        //æ¯æ¡çº¿ç¨‹è´Ÿè´£ä¸€ä¸ªé€‰æ‹©å™¨çš„æŸ¥è¯¢
        final Selector selector;

            this.selector = selector;

        public void run() {
            try {
                while (!Thread.interrupted()) {
                    //å•ä½ä¸ºæ¯«ç§’
                    selector.select(1000);
                    Set<SelectionKey> selectedKeys = selector.selectedKeys();
                    if (null == selectedKeys || selectedKeys.size() == 0) {
                        continue;

                    Iterator<SelectionKey> it = selectedKeys.iterator();
                    while (it.hasNext()) {
                        //Reactorè´Ÿè´£dispatchæ”¶åˆ°çš„äº‹ä»¶
                        SelectionKey sk = it.next();

                ex.printStackTrace();

        }


        void dispatch(SelectionKey sk) {
            Runnable handler = (Runnable) sk.attachment();
            //è°ƒç”¨ä¹‹å‰attachç»‘å®šåˆ°é€‰æ‹©é”®çš„handlerå¤„ç†å™¨å¯¹è±¡

    class AcceptorHandler implements Runnable {
        public void run() {
            try {
                SocketChannel channel = serverSocket.accept();
                Logger.info("æ¥æ”¶åˆ°ä¸€ä¸ªæ–°çš„è¿æ¥");

                e.printStackTrace();
            }
            if (next.incrementAndGet() == workSelectors.length) {
                next.set(0);
            }
        }
    }

    public static void main(String[] args) throws IOException {
        MultiThreadEchoServerReactor server =
                new MultiThreadEchoServerReactor();
        server.startService();
    }
